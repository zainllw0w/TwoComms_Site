def get_description_for_feed(product) -> tuple:
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç (description, rich_text_description) –¥–ª—è —Ñ–∏–¥–∞.
    
    –ü—Ä–∏–æ—Ä–∏—Ç–µ—Ç:
    1. rich_text_description (–µ—Å–ª–∏ –µ—Å—Ç—å full_description —Å HTML)
    2. description (plain text)
    3. short_description
    4. –î–µ—Ñ–æ–ª—Ç–Ω–æ–µ –æ–ø–∏—Å–∞–Ω–∏–µ
    """
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º full_description (–º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å HTML)
    if product.full_description:
        # –ï—Å–ª–∏ —Å–æ–¥–µ—Ä–∂–∏—Ç HTML —Ç–µ–≥–∏ - –∏—Å–ø–æ–ª—å–∑—É–µ–º –∫–∞–∫ rich_text_description
        if '<' in product.full_description and '>' in product.full_description:
            return (product.description or product.short_description or '', product.full_description)
        else:
            return (product.full_description, None)
    
    # –ò—Å–ø–æ–ª—å–∑—É–µ–º description –∏–ª–∏ short_description
    description = product.description or product.short_description or ''
    return (description, None)
